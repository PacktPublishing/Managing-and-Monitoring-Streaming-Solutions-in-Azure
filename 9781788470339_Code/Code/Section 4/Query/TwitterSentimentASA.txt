SELECT Topic,count(*) AS Count, Avg(SentimentScore) AS AvgSentiment, 
System.Timestamp AS Insert_Time
INTO TweetBI
FROM TwitterInput TIMESTAMP BY CreatedAt 
GROUP BY HoppingWindow(second,10,5), Topic
